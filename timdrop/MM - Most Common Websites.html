num_page = 4

C = Counter()
for CTN in ct_no:
    li = [x.link for x in google.search(CTN, num_page)]
    ui = []
    for x in li:
        try:
            ui.append(urlparse(x))
        except AttributeError:
            pass
    bi = [x.netloc.encode('utf-8') for x in ui]
    C.update(bi)

C_mc = C.most_common(15)
top_sites = []
for i in range(len(C_mc)):
    top_sites.append(C_mc[i][0])
    
df = pd.DataFrame(top_sites, columns = ['Top Sites'])
df.to_csv('D:/Python_Database/Myeloma/Top_Sites.csv')
